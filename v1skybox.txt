import * as THREE from 'three'
import { OrbitControls } from './jsm/controls/OrbitControls.js'
//escena
const scene = new THREE.Scene()

//camara
const camera = new THREE.PerspectiveCamera(55, window.innerWidth / window.innerHeight, 0.1, 1000)
camera.position.set(-900, -200,-900)

// Create a WebGLRenderer and turn on shadows in the renderer
const renderer = new THREE.WebGLRenderer()
renderer.setSize(window.innerWidth, window.innerHeight)
document.body.appendChild(renderer.domElement)

const controls = new OrbitControls(camera, renderer.domElement)
controls.addEventListener('change', renderer)
controls.minDistance = 500
controls.maxDistance = 1500

let materialArray = [];
const texture_ft = new THREE.TextureLoader().load( 'img/space_ft.png');
const texture_bk = new THREE.TextureLoader().load( 'img/space_bk.png');
const texture_up = new THREE.TextureLoader().load( 'img/space_up.png');
const texture_dn = new THREE.TextureLoader().load( 'img/space_dn.png');
const texture_rt = new THREE.TextureLoader().load( 'img/space_rt.png');
const texture_lf = new THREE.TextureLoader().load( 'img/space_lf.png');
  
materialArray.push(new THREE.MeshBasicMaterial( { map: texture_ft }));
materialArray.push(new THREE.MeshBasicMaterial( { map: texture_bk }));
materialArray.push(new THREE.MeshBasicMaterial( { map: texture_up }));
materialArray.push(new THREE.MeshBasicMaterial( { map: texture_dn }));
materialArray.push(new THREE.MeshBasicMaterial( { map: texture_rt }));
materialArray.push(new THREE.MeshBasicMaterial( { map: texture_lf }));
   
for (let i = 0; i < 6; i++)
  materialArray[i].side = THREE.BackSide;
   
let skyboxGeo = new THREE.BoxGeometry( 10000, 10000, 10000);
let skybox = new THREE.Mesh( skyboxGeo, materialArray );
scene.add( skybox );
